# Folder containing your images
img_dir <- "ppt_pics2"

# 1. List all image files (PNG/JPG)
img_files <- list.files(
  path    = img_dir,
  pattern = "\\.(png|jpe?g)$",
  full.names = TRUE
)

if (length(img_files) == 0) {
  stop("No PNG/JPG files found in folder: ", img_dir)
}

# 2. Build data frame for sorting and titles
df_imgs <- data.frame(
  path = img_files,
  file = basename(img_files),
  stringsAsFactors = FALSE
)

# Extract prefix before first underscore (e.g. "9", "9b", "14", etc.)
df_imgs$prefix <- sub("_.*$", "", df_imgs$file)

# Numeric part: leading digits
df_imgs$number <- as.integer(str_extract(df_imgs$prefix, "^[0-9]+"))

# Suffix: whatever is left after leading digits (no lookbehind)
df_imgs$suffix <- sub("^[0-9]+", "", df_imgs$prefix)
df_imgs$suffix[is.na(df_imgs$suffix)] <- ""

# Sort first by number, then by suffix, then by filename
df_imgs <- df_imgs[order(df_imgs$number, df_imgs$suffix, df_imgs$file), ]

# 3. Build slide titles
#    - Drop everything before the first underscore
#    - Drop extension
#    - Replace underscores with spaces
#    - Title-case each word


df_imgs$title_raw <- ifelse(
  grepl("_", df_imgs$file),
  sub("^[^_]*_", "", df_imgs$file),  # keep after first underscore
  df_imgs$file
)

df_imgs$title_raw <- sub("\\.[^.]+$", "", df_imgs$title_raw)  # drop extension
df_imgs$title_raw <- gsub("_", " ", df_imgs$title_raw)        # underscores → spaces

df_imgs$title <- str_to_title(df_imgs$title_raw)

# 4. Create PowerPoint and add one slide per image
ppt <- read_pptx()

# If you want to see available layouts:
# layout_summary(ppt)

for (i in seq_len(nrow(df_imgs))) {
  ppt <- add_slide(ppt, layout = "Title and Content", master = "Office Theme")
  
  # Add title
  ppt <- ph_with(
    ppt,
    value    = df_imgs$title[i],
    location = ph_location_type(type = "title")
  )
  
  # Add picture (body placeholder)
  ppt <- ph_with(
    ppt,
    value    = external_img(df_imgs$path[i]),
    location = ph_location_type(type = "body")
  )
}
